""" 小风哥 Telegram Bot V2（个性化版） 功能： 1. 网页表单提交时，自动 POST 到本 Bot 的 /webhook 接口 2. Bot 把用户信息存起来（按用户的 Telegram 用户名为 key） 3. 用户 /start 时，Bot 认出他，发个性化欢迎语 + 4 个按钮 4. 4 个按钮：VIP 群 / 跟单 & iPad / 早鸟配套 / 裂变活动 依赖： pip install python-telegram-bot flask --upgrade 运行： python xiaofengge_bot_v2.py (会同时启动 Bot 和一个 Web 服务在 5000 端口) """ import os import json import asyncio import threading from flask import Flask, request, jsonify from telegram import Update, InlineKeyboardButton, InlineKeyboardMarkup from telegram.ext import ( Application, CommandHandler, CallbackQueryHandler, ContextTypes, ) # ========== 配置区 ========== BOT_TOKEN = os.environ.get("BOT_TOKEN", "在这里贴上 BotFather 给你的 TOKEN") WEBHOOK_SECRET = os.environ.get("WEBHOOK_SECRET", "随便一串密码_用来防止陌生人乱提交") ADMIN_CHAT_ID = os.environ.get("ADMIN_CHAT_ID", "") # 你自己的 Telegram chat_id（接收通知） PORT = int(os.environ.get("PORT", 5000)) DATA_FILE = "users.json" # 用户数据持久化到这个文件 # 哪些按钮被点了要通知 admin？ NOTIFY_BUTTONS = {"vip", "early_bird"} # VIP 群 + 早鸟配套 # 4 个按钮的内容 CONTENT = { "vip": { "type": "photo_with_link", "photo": "https://你的图片链接.jpg", "caption": ( "👑 *VIP 群福利*\n\n" "✅ 每日精选爆料\n" "✅ 独家投资策略\n" "✅ 24 小时客服\n" "✅ 资深导师指导\n\n" "立即加入，享受顶级会员待遇！" ), "button_text": "🚀 立即加入 VIP 群", "button_url": "https://t.me/+你的VIP群链接", }, "follow_ipad": { "type": "photo_with_link", "photo": "https://你的图片链接2.jpg", "caption": ( "📱 *跟单活动 & iPad 大奖*\n\n" "🎯 跟单赢取 iPad Pro\n" "💰 跟随导师操作\n" "🏆 月度排行榜奖励\n\n" "活动时间：即日起至月底\n" "点击下方按钮参加活动 👇" ), "button_text": "🎁 参加活动", "button_url": "https://你的活动链接.com", }, "early_bird": { "type": "text_with_link", "text": ( "🐦 *早鸟配套*\n\n" "限时优惠：原价 ¥999 → 现价 ¥299\n\n" "📦 配套内容：\n" "• 一对一导师服务\n" "• 内部分析报告\n" "• 专属交流群\n" "• 7 天试用期\n\n" "⏰ 仅限前 100 名！" ), "button_text": "💳 立即抢购", "button_url": "https://你的购买链接.com", }, "viral": { "type": "text_only", "text": ( "🔥 *裂变活动*\n\n" "邀请好友，双方都有奖！\n\n" "🎁 邀请奖励：\n" "• 邀请 1 人：¥10 现金\n" "• 邀请 3 人：¥50 + VIP 体验\n" "• 邀请 10 人：iPhone 15 抽奖资格" ), "button_text": "📤 分享", "button_url": "https://t.me/share/url?url=https://你的网站", }, } # ============================= # -------- 用户数据存储 -------- def load_users(): if os.path.exists(DATA_FILE): with open(DATA_FILE, "r", encoding="utf-8") as f: return json.load(f) return {} def save_users(users): with open(DATA_FILE, "w", encoding="utf-8") as f: json.dump(users, f, ensure_ascii=False, indent=2) users_db = load_users() # { "@username": { "name": ..., "phone": ..., ... } } # -------- Flask 接收网页表单 -------- flask_app = Flask(__name__) @flask_app.route("/webhook", methods=["POST"]) def receive_form(): # 简单的密码验证 secret = request.headers.get("X-Secret") or request.json.get("secret") if secret != WEBHOOK_SECRET: return jsonify({"error": "unauthorized"}), 401 data = request.json # 假设网页表单 POST 过来的内容是： # { "name": "...", "phone": "...", "telegram_id": "@username", ... } tg_id = (data.get("telegram_id") or "").strip().lstrip("@").lower() if not tg_id: return jsonify({"error": "missing telegram_id"}), 400 users_db[tg_id] = { "name": data.get("name", ""), "phone": data.get("phone", ""), "email": data.get("email", ""), "submitted_at": data.get("submitted_at", ""), } save_users(users_db) print(f"✅ 收到新用户：@{tg_id} -> {users_db[tg_id]}") return jsonify({"ok": True}) @flask_app.route("/health", methods=["GET"]) def health(): return jsonify({"status": "ok", "users": len(users_db)}) def run_flask(): flask_app.run(host="0.0.0.0", port=PORT, debug=False, use_reloader=False) # -------- Telegram Bot -------- def build_main_menu(): return InlineKeyboardMarkup([ [ InlineKeyboardButton("👑 VIP 群", callback_data="vip"), InlineKeyboardButton("📱 跟单 & iPad", callback_data="follow_ipad"), ], [ InlineKeyboardButton("🐦 早鸟配套", callback_data="early_bird"), InlineKeyboardButton("🔥 裂变活动", callback_data="viral"), ], ]) async def start(update: Update, context: ContextTypes.DEFAULT_TYPE): user = update.effective_user username = (user.username or "").lower() first_name = user.first_name or "朋友" # 把用户的 /start 讯息 ID 记下来（等下要转发给 admin） context.user_data["start_message_id"] = update.message.message_id context.user_data["start_chat_id"] = update.effective_chat.id # 看看是不是表单提交过的用户 info = users_db.get(username) if info: # 个性化欢迎 welcome_text = ( f"👋 {first_name}，欢迎回来！\n\n" f"✅ 已成功接收你的报名资料：\n" f"姓名：{info.get('name', '—')}\n" f"电话：{info.get('phone', '—')}\n\n" "我们的客服稍后会主动联系你。\n" "在这之前，先看看以下服务吧 👇" ) else: # 普通欢迎 welcome_text = ( f"👋 你好 *{first_name}*！\n\n" "🎉 欢迎加入小风哥团队！\n" "请选择你感兴趣的服务 👇\n\n" "_(如果你刚刚在网站提交了资料，请确认 Telegram 用户名已经设定哦)_" ) await update.message.reply_text( welcome_text, reply_markup=build_main_menu(), parse_mode="Markdown", ) async def notify_admin(context, query, data): """转发用户的 /start 讯息给 admin，并附上小提示""" if not ADMIN_CHAT_ID: return # 没设 admin 就不通知 button_name = { "vip": "👑 VIP 群", "early_bird": "🐦 早鸟配套", "follow_ipad": "📱 跟单 & iPad", "viral": "🔥 裂变活动", }.get(data, data) start_msg_id = context.user_data.get("start_message_id") start_chat_id = context.user_data.get("start_chat_id") user = query.from_user username_line = f"@{user.username}" if user.username else "（无用户名）" try: # 先发简短通知（含 Telegram ID） await context.bot.send_message( chat_id=ADMIN_CHAT_ID, text=( f"🔔 用户点击了：{button_name}\n" f"👤 Telegram：{username_line}\n" f"🆔 ID：`{user.id}`" ), parse_mode="Markdown", ) # 再 forward 用户的 /start 讯息 if start_msg_id and start_chat_id: await context.bot.forward_message( chat_id=ADMIN_CHAT_ID, from_chat_id=start_chat_id, message_id=start_msg_id, ) except Exception as e: print(f"⚠️ 通知 admin 失败：{e}") async def myid(update: Update, context: ContextTypes.DEFAULT_TYPE): """显示自己的 chat_id，用来设 ADMIN_CHAT_ID""" user = update.effective_user chat_id = update.effective_chat.id await update.message.reply_text( f"你的 chat_id 是：`{chat_id}`\n\n" f"把这个数字贴到 Railway 的 `ADMIN_CHAT_ID` 环境变数里。", parse_mode="Markdown", ) async def button_handler(update: Update, context: ContextTypes.DEFAULT_TYPE): query = update.callback_query await query.answer() # 立即响应，按钮停止转圈 data = query.data if data not in CONTENT: return # 🔔 如果点的是「热单」按钮 → 通知 admin if data in NOTIFY_BUTTONS: await notify_admin(context, query, data) item = CONTENT[data] chat_id = query.message.chat_id # 「返回主菜单」按钮 back_btn = [InlineKeyboardButton("⬅️ 返回主菜单", callback_data="back_to_menu")] if item["type"] == "photo_with_link": markup = InlineKeyboardMarkup([ [InlineKeyboardButton(item["button_text"], url=item["button_url"])], back_btn, ]) # 先删掉旧讯息，再发新的图片讯息（图片不能用 edit 改成文字） try: await query.message.delete() except Exception: pass await context.bot.send_photo( chat_id=chat_id, photo=item["photo"], caption=item["caption"], parse_mode="Markdown", reply_markup=markup, ) elif item["type"] == "text_with_link": markup = InlineKeyboardMarkup([ [InlineKeyboardButton(item["button_text"], url=item["button_url"])], back_btn, ]) # 直接编辑现有讯息（瞬间响应！） try: await query.edit_message_text( text=item["text"], parse_mode="Markdown", reply_markup=markup, ) except Exception: # 如果上一则是图片（不能 edit 成文字），改成发新讯息 await context.bot.send_message( chat_id=chat_id, text=item["text"], parse_mode="Markdown", reply_markup=markup, ) elif item["type"] == "text_only": markup = InlineKeyboardMarkup([back_btn]) try: await query.edit_message_text( text=item["text"], parse_mode="Markdown", reply_markup=markup, ) except Exception: await context.bot.send_message( chat_id=chat_id, text=item["text"], parse_mode="Markdown", reply_markup=markup, ) async def back_to_menu(update: Update, context: ContextTypes.DEFAULT_TYPE): query = update.callback_query await query.answer() chat_id = query.message.chat_id menu_text = "📋 主菜单 —— 请选择服务：" try: await query.edit_message_text( text=menu_text, reply_markup=build_main_menu(), ) except Exception: # 上一则是图片时 edit 不了，改发新讯息 try: await query.message.delete() except Exception: pass await context.bot.send_message( chat_id=chat_id, text=menu_text, reply_markup=build_main_menu(), ) async def post_init(application): """Bot 启动后立即执行 —— 清掉旧的 menu / commands""" # 1. 清空所有 / 指令清单 await application.bot.delete_my_commands() # 2. 把 menu button 改回 default（不显示 menu 按钮） await application.bot.set_chat_menu_button(menu_button={"type": "default"}) print("✅ 已清空旧的 commands 和 menu button") def main(): # 后台跑 Flask，主线程跑 Bot flask_thread = threading.Thread(target=run_flask, daemon=True) flask_thread.start() print(f"✅ Webhook 服务运行在 :{PORT}/webhook") app = ( Application.builder() .token(BOT_TOKEN) .post_init(post_init) # 启动后清菜单 .build() ) app.add_handler(CommandHandler("start", start)) app.add_handler(CommandHandler("myid", myid)) # 取得自己 chat_id 的指令 app.add_handler(CallbackQueryHandler(back_to_menu, pattern="^back_to_menu$")) app.add_handler(CallbackQueryHandler(button_handler)) print("✅ Bot 启动成功！按 Ctrl+C 停止") app.run_polling() if __name__ == "__main__": main()
